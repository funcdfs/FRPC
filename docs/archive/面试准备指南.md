**代码位置**:
```

```

**关键代码片段**:
```cpp
// 事件循环核心逻辑
void NetworkEngine::run() {
    while (running_) {
        // 1. epoll_wait 等待事件
        int nfds = epoll_wait(epoll_fd_, events, MAX_EVENTS, timeout);
        
        // 2. 遍历就绪事件
        for (int i = 0; i < nfds; ++i) {
            int fd = events[i].data.fd;
            
            // 3. 分发到对应的处理器
            if (events[i].events & EPOLLIN) {
                handlers_[fd].on_read();
            }
            if (events[i].events & EPOLLOUT) {
                handlers_[fd].on_write();
            }
        }
    }
}
```

**回答要点**:
- Reactor 是同步非阻塞 I/O，应用程序主动读写
- Proactor 是异步 I/O，操作系统完成读写后通知
- 选择 Reactor 因为 Linux epoll 成熟，控制粒度细
- 使用 EPOLLET（边缘触发）减少事件通知次数

**关键代码片段**:
```cpp
// Awaitable 对象封装异步发送
struct SendAwaitable {
    // 1. 检查是否已完成
    bool await_ready() const noexcept {
        // 尝试立即发送
        ssize_t sent = ::send(fd_, data_.data(), data_.size(), MSG_DONTWAIT);
        if (sent == data_.size()) {
            completed_ = true;
            return true;  // 已完成，无需挂起
        }
        return false;  // 需要挂起
    }
    
    // 2. 挂起协程
    void await_suspend(std::coroutine_handle<> handle) {
        handle_ = handle;
        // 注册写事件，当 socket 可写时恢复协程
        engine_->register_write_event(fd_, [this]() {
            // 发送数据
            ssize_t sent = ::send(fd_, data_.data(), data_.size(), 0);
            result_ = SendResult{sent > 0, sent};
            completed_ = true;
            // 恢复协程
            CoroutineScheduler::instance().resume(handle_);
        });
    }
    
    // 3. 获取结果
    SendResult await_resume() const {
        return result_;
    }
};

// 用户代码
RpcTask<void> send_data() {
    // 同步风格编写异步代码
    auto result = co_await engine.async_send(fd, data);
    if (result.success) {
        // 处理成功
    }
}
```

**回答要点**:
- 三要素：Promise Type、Coroutine Handle、Awaitable Object
- co_await 流程：await_ready → await_suspend → [挂起] → [恢复] → await_resume
- 通过 Awaitable 对象封装异步操作，注册回调恢复协程
- 协程帧保存在堆上（或内存池），包含局部变量和执行位置

```cpp
// 传统回调方式 - 回调地狱
void send_and_recv(int fd, Callback callback) {
    async_send(fd, data, [fd, callback](SendResult result) {
        if (!result.success) {
            callback(Error{"Send failed"});
            return;
        }
        async_recv(fd, [callback](RecvResult result) {
            if (!result.success) {
                callback(Error{"Recv failed"});
                return;
            }
            callback(result.data);
        });
    });
}

// 协程方式 - 清晰简洁
RpcTask<Data> send_and_recv(int fd) {
    auto send_result = co_await async_send(fd, data);
    if (!send_result.success) {
        throw SendException("Send failed");
    }
    
    auto recv_result = co_await async_recv(fd);
    if (!recv_result.success) {
        throw RecvException("Recv failed");
    }
    
    co_return recv_result.data;
}
```

- 调试友好：堆栈连续，易于追踪

```cpp
// Promise Type 自定义内存分配
template<typename T>
struct RpcTaskPromise {
    // 使用内存池分配
    void* operator new(size_t size) {
        return CoroutineScheduler::instance().allocate_frame(size);
    }
    
    void operator delete(void* ptr, size_t size) {
        CoroutineScheduler::instance().deallocate_frame(ptr, size);
    }
};

// 内存池实现
class MemoryPool {
public:
    void* allocate(size_t size) {
        std::lock_guard lock(mutex_);
        if (free_list_.empty()) {
            // 扩容
            return new uint8_t[size];
        }
        void* ptr = free_list_.top();
        free_list_.pop();
        return ptr;
    }
    
    void deallocate(void* ptr) {
        std::lock_guard lock(mutex_);
        free_list_.push(ptr);
    }

private:
    std::stack<void*> free_list_;  // 空闲链表
    std::mutex mutex_;
};
```

**回答要点**:

- 内存池策略：预分配、空闲链表、线程本地缓存
- 性能提升：分配开销降低 90%（~10ns），减少碎片
- 测试数据：减少 80% 的堆分配次数

```cpp
RpcTask<void> HealthChecker::check_loop() {
    while (running_) {
        for (auto& target : targets_) {
            bool healthy = co_await check_instance(target.instance);
            
            if (healthy) {
                target.consecutive_failures = 0;
                if (++target.consecutive_successes >= config_.success_threshold) {
                    // 标记为健康
                    registry_->update_health_status(
                        target.service_name, target.instance, true);
                }
            } else {
                target.consecutive_successes = 0;
                if (++target.consecutive_failures >= config_.failure_threshold) {
                    // 标记为不健康
                    registry_->update_health_status(
                        target.service_name, target.instance, false);
                }

        }

    }
}
```

- 连续成功 M 次（默认 2 次）恢复为健康
- 不健康实例从服务列表移除，恢复后重新添加

```cpp
// 轮询策略
class RoundRobinLoadBalancer : public LoadBalancer {
public:
    ServiceInstance select(const std::vector<ServiceInstance>& instances) override {
        if (instances.empty()) {
            throw NoInstanceAvailableException();
        }
        
        // 原子递增，线程安全
        size_t index = index_.fetch_add(1, std::memory_order_relaxed);
        return instances[index % instances.size()];
    }

private:
    std::atomic<size_t> index_{0};
};

// 加权轮询策略
class WeightedRoundRobinLoadBalancer : public LoadBalancer {
public:
    ServiceInstance select(const std::vector<ServiceInstance>& instances) override {
        int total_weight = 0;
        for (const auto& inst : instances) {
            total_weight += inst.weight;
        }
        
        int current = current_weight_.fetch_add(1, std::memory_order_relaxed);
        int target = current % total_weight;
        
        int sum = 0;
        for (const auto& inst : instances) {
            sum += inst.weight;
            if (target < sum) {
                return inst;
            }
        }
        return instances[0];
    }

private:
    std::atomic<int> current_weight_{0};
};
```

**回答要点**:
- 支持：轮询、随机、最少连接、加权轮询
- 使用 atomic 保证线程安全，无锁设计
- 加权轮询按权重比例分配流量
- 选择策略：实例性能相近用轮询，差异大用加权

```cpp
// 获取连接
RpcTask<Connection> ConnectionPool::get_connection(const ServiceInstance& instance) {
    std::unique_lock lock(mutex_);
    
    auto& pool = pools_[instance];
    
    // 1. 优先返回空闲连接
    if (!pool.idle_connections.empty()) {
        auto conn = std::move(pool.idle_connections.back());
        pool.idle_connections.pop_back();
        pool.active_connections.push_back(conn);
        co_return conn;
    }
    
    // 2. 创建新连接（如果未达上限）
    if (pool.total_count < config_.max_connections) {
        auto conn = co_await create_connection(instance);
        pool.active_connections.push_back(conn);
        pool.total_count++;
        co_return conn;
    }
    
    // 3. 等待或报错
    throw ResourceExhaustedException("Connection pool full");
}

// 归还连接
void ConnectionPool::return_connection(Connection conn) {
    std::unique_lock lock(mutex_);
    
    if (!conn.is_valid()) {
        // 连接失效，移除
        pool.total_count--;
        return;
    }
    
    // 移到空闲列表
    pool.idle_connections.push_back(std::move(conn));
}
```

**回答要点**:
- TCP 握手开销大（1-3ms），连接池复用连接
- 管理：最小/最大连接数、空闲超时、健康检测
- 失效处理：检测到错误立即移除，不放回池中
- 复用率：>90%，显著降低握手开销
